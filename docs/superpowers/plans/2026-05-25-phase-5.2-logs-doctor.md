### Step 1.2: impl

```swift
import Foundation

public enum RelativeTimeError: Error, Equatable, LocalizedError {
    case invalidFormat(String)
    public var errorDescription: String? {
        switch self {
        case .invalidFormat(let s): return "Invalid time spec: \(s) (expected Nm|Nh|Nd|Ns or ISO 8601)"
        }
    }
}

/// Parses either a relative duration (`5m`, `2h`, `1d`, `90s`) — returned
/// as a `Date` in the past — or an ISO 8601 timestamp.
public enum RelativeTime {
    public static func parse(_ input: String, relativeTo now: Date) throws -> Date {
        guard !input.isEmpty else { throw RelativeTimeError.invalidFormat(input) }
        if let last = input.last, "smhd".contains(last) {
            let numPart = String(input.dropLast())
            guard let n = Int(numPart), n > 0 else { throw RelativeTimeError.invalidFormat(input) }
            let multiplier: TimeInterval = {
                switch last {
                case "s": return 1
                case "m": return 60
                case "h": return 3_600
                case "d": return 86_400
                default: return 0
                }
            }()
            return now.addingTimeInterval(-Double(n) * multiplier)
        }
        let iso = ISO8601DateFormatter()
        if let parsed = iso.date(from: input) {
            return parsed
        }
        throw RelativeTimeError.invalidFormat(input)
    }
}
```

```swift
        var capturedResp: URLResponse?
        var capturedErr: Error?

            capturedData = d
            capturedResp = r
            capturedErr = e
            expectation.fulfill()
        }.resume()
        wait(for: [expectation], timeout: 5)
        if let err = capturedErr { throw err }
        return (capturedData ?? Data(), capturedResp!)
    }
}
```

## Task 3 : `Sources/iris/Support/SignalHandling.swift` — SIGINT trap

**Files:** `Sources/iris/Support/SignalHandling.swift`

**Pourquoi** : `iris logs --follow` doit pouvoir être interrompu par Ctrl-C et clean-up (close SSE connection, exit 0). Sans trap, SIGINT laisse le process en zombie ou exit 130 brutal.

### Step 3.1: impl

```swift
import Foundation
import Darwin

/// Installs a one-shot SIGINT handler that invokes the provided callback
/// asynchronously. Returns a token that, when held, keeps the handler
/// armed; drop it to restore default SIGINT behavior.
///
/// Note : `signal(3)` is the simplest POSIX API but is async-signal-safe
/// only for a narrow set of operations. We use it as a wake signal — the
/// handler sets an atomic flag and dispatches to a normal queue where the
/// callback runs (no NIO/Swift Concurrency primitives inside the signal
/// handler itself).
enum SignalHandling {
    /// Install a SIGINT handler. The callback fires on the next signal
    /// arrival. Subsequent SIGINTs propagate to the default action
    /// (terminate immediately) — useful for "Ctrl-C twice to force quit".
    static func onSIGINTOnce(_ callback: @escaping @Sendable () -> Void) {
        let box = SignalBox(callback: callback)
        sigintBox = box
        signal(SIGINT) { _ in
            let box = sigintBox
            sigintBox = nil
            signal(SIGINT, SIG_DFL)
            if let box = box {
                DispatchQueue.global(qos: .userInitiated).async {
                    box.callback()
                }
            }
        }
    }
}

private final class SignalBox: @unchecked Sendable {
    let callback: @Sendable () -> Void
    init(callback: @escaping @Sendable () -> Void) {
        self.callback = callback
    }
}

private nonisolated(unsafe) var sigintBox: SignalBox?
```

```swift
import ArgumentParser
import Foundation
import IrisKit

struct LogsCommand: AsyncParsableCommand {
    static let configuration = CommandConfiguration(
        commandName: "logs",
        abstract: "Query daemon events (one-shot or --follow)."
    )

    @OptionGroup var connection: ConnectionOptions

    @Option(name: .customLong("since"), help: "Lower bound: ISO 8601 or relative (5m, 1h, 1d).")
    var since: String?

    @Option(name: .customLong("until"), help: "Upper bound: ISO 8601 or relative.")
    var until: String?

    @Option(name: .customLong("limit"), help: "Max events to return (one-shot only).")
    var limit: Int?

    @Option(name: .customLong("kind"), help: "Comma-separated event kinds (substituted, no_match, error, pass_through).")
    var kindRaw: String = ""

    @Option(name: .customLong("host"), help: "Filter by host.")
    var host: String?

    @Flag(name: .customLong("follow"), help: "Stream events via SSE until SIGINT.")
    var follow: Bool = false

    @Flag(name: .customLong("json"), help: "Emit JSON (ndjson in --follow mode).")
    var json: Bool = false

    mutating func validate() throws {
        if follow && (since != nil || until != nil || limit != nil) {
            throw ValidationError("--follow is incompatible with --since/--until/--limit")
        }
    }

    mutating func run() async throws {
        let kinds: [Event.Kind] = try kindRaw
            .split(separator: ",", omittingEmptySubsequences: true)
            .map(String.init)
            .map { raw in
                guard let k = Event.Kind(rawValue: raw) else {
                    throw ValidationError("unknown event kind: \(raw)")
                }
                return k
            }

        if follow {
            try await runFollow(kinds: kinds)
        } else {
            try await runOneShot(kinds: kinds)
        }
    }

    private func runOneShot(kinds: [Event.Kind]) async throws {
        let now = Date()
        let sinceDate = try since.map { try RelativeTime.parse($0, relativeTo: now) }
        let untilDate = try until.map { try RelativeTime.parse($0, relativeTo: now) }
        let events = try await withAdminClient(connection) { client in
            try await client.call(
                .eventsQuery,
                params: EventsQueryParams(
                    since: sinceDate,
                    until: untilDate,
                    limit: limit,
                    kind: kinds.isEmpty ? nil : kinds,
                    host: host
                ),
                returning: [Event].self
            )
        }
        if json {
            let encoder = JSONEncoder()
            encoder.outputFormatting = [.prettyPrinted, .sortedKeys, .withoutEscapingSlashes]
            encoder.dateEncodingStrategy = .iso8601
            let data = try encoder.encode(events)
            print(String(data: data, encoding: .utf8) ?? "[]")
        } else {
            if events.isEmpty {
                print("no events")
            } else {
                for event in events {
                    print(formatEvent(event))
                }
            }
        }
    }

    private func runFollow(kinds: [Event.Kind]) async throws {
        let path = try connection.resolvedSocketPath()
        let configPath = connection.configPath
        // EventsClient needs the events SSE endpoint URL. Derive it from
        // config (we already have admin socket; the config_get call also
        // returns broker.events_listen).
        let eventsURL = try await withAdminClient(connection) { client in
            let config = try await client.call(.configGet, returning: Config.self)
            return URL(string: "http://\(config.broker.eventsListen ?? "")/events")!
        }

        let client = EventsClient(url: eventsURL)
        // Install SIGINT handler that cancels the stream
        SignalHandling.onSIGINTOnce {
            client.cancel()
        }

        let kindSet = Set(kinds)
        let encoder = JSONEncoder()
        encoder.outputFormatting = [.sortedKeys, .withoutEscapingSlashes]
        encoder.dateEncodingStrategy = .iso8601

        do {
            for try await event in client.events() {
                if !kindSet.isEmpty && !kindSet.contains(event.kind) { continue }
                if let h = host, event.host != h { continue }
                if json {
                    if let data = try? encoder.encode(event),
                       let line = String(data: data, encoding: .utf8) {
                        print(line)
                    }
                } else {
                    print(formatEvent(event))
                }
            }
        } catch {
            FileHandle.standardError.write(Data("stream disconnected: \(error)\n".utf8))
            throw ExitCode(IrisExitCode.ioError)
        }
    }

    private func formatEvent(_ event: Event) -> String {
        let iso = ISO8601DateFormatter().string(from: event.timestamp)
        let host = event.host ?? "-"
        let method = event.method ?? "-"
        let path = event.path ?? "-"
        let dur = event.durationMs.map { "\($0)ms" } ?? "-"
        let subs = event.substitutions?.joined(separator: ",") ?? "-"
        return "\(iso)  \(event.kind.rawValue)  \(host)  \(method) \(path)  sub=\(subs) dur=\(dur)"
    }
}
```

### Step 5.1: impl

```swift
import ArgumentParser
import Foundation
import IrisKit

struct DoctorCommand: AsyncParsableCommand {
    static let configuration = CommandConfiguration(
        commandName: "doctor",
        abstract: "Run health checks against the daemon and local environment."
    )

    @OptionGroup var connection: ConnectionOptions
    @Flag(name: .customLong("json")) var json: Bool = false

    enum Severity: String, Codable, Sendable {
        case ok, warn, fail
    }

    struct CheckResult: Codable, Sendable {
        let name: String
        let severity: Severity
        let detail: String
    }

    mutating func run() async throws {
        var results: [CheckResult] = []

        // Resolve socket path early (used by check 1)
        let socketPath: String
        do {
            socketPath = try connection.resolvedSocketPath()
        } catch {
            results.append(.init(name: "socket-path-resolution", severity: .fail, detail: "\(error)"))
            emit(results)
            throw ExitCode(IrisExitCode.logicError)
        }

        // Check 1: socket file exists
        if FileManager.default.fileExists(atPath: socketPath) {
            results.append(.init(name: "admin-socket-present", severity: .ok, detail: socketPath))
        } else {
            results.append(.init(name: "admin-socket-present", severity: .fail, detail: "missing: \(socketPath)"))
            emit(results)
            throw ExitCode(IrisExitCode.logicError)
        }

        // Checks 2 + 3 + 4 + 6 require the daemon. If any RPC fails, we mark
        // them all fail but keep going to check env vars (check 5) and
        // apiKeyHelper (check 7).
        var status: DaemonStatus?
        var caPath: String?
        var caTrusted: Bool?
        var brokerListen: String?

        do {
            try await withAdminClient(connection) { client in
                status = try await client.call(.daemonStatus, returning: DaemonStatus.self)
                let p = try await client.call(.caExportPath, returning: CAExportPathResult.self)
                caPath = p.path
                let t = try await client.call(.caIsTrusted, returning: CAIsTrustedResult.self)
                caTrusted = t.trusted
                let cfg = try await client.call(.configGet, returning: Config.self)
                brokerListen = cfg.broker.listen
            }
        } catch {
            results.append(.init(name: "daemon-rpc", severity: .fail, detail: "\(error)"))
            emit(results)
            throw ExitCode(IrisExitCode.logicError)
        }

        // Check 2: daemon alive (kill -0)
        if let st = status {
            let alive = kill(st.pid, 0) == 0
            results.append(.init(
                name: "daemon-alive",
                severity: alive ? .ok : .fail,
                detail: "pid=\(st.pid) uptime=\(TextFormatter.uptime(seconds: st.uptimeS))"
            ))
        }

        // Check 3: CA cert file present
        if let p = caPath {
            if FileManager.default.fileExists(atPath: p) {
                results.append(.init(name: "ca-cert-present", severity: .ok, detail: p))
            } else {
                results.append(.init(name: "ca-cert-present", severity: .fail, detail: "missing: \(p)"))
            }
        }

        // Check 4: CA trusted system-wide
        if let t = caTrusted {
            results.append(.init(
                name: "ca-trusted-system",
                severity: t ? .ok : .fail,
                detail: t ? "in user trust store" : "not in user trust store"
            ))
        }

        // Check 5: env vars
        let env = ProcessInfo.processInfo.environment
        let expectedVars = ["HTTPS_PROXY", "HTTP_PROXY", "NODE_EXTRA_CA_CERTS", "SSL_CERT_FILE"]
        let missing = expectedVars.filter { env[$0] == nil || env[$0]?.isEmpty == true }
        if missing.isEmpty {
            results.append(.init(name: "shell-env-vars", severity: .ok, detail: "all set"))
        } else {
            results.append(.init(name: "shell-env-vars", severity: .warn, detail: "missing: \(missing.joined(separator: ","))"))
        }

        // Check 6: ping endpoint
        if let listen = brokerListen {
            let pingURL = URL(string: "http://\(listen)/__iris_ping")!
            let pingResult: Bool = await {
                await withCheckedContinuation { continuation in
                    var req = URLRequest(url: pingURL)
                    req.timeoutInterval = 2
                    URLSession.shared.dataTask(with: req) { data, resp, _ in
                        let http = resp as? HTTPURLResponse
                        let ok = http?.statusCode == 200 && String(data: data ?? Data(), encoding: .utf8) == "ok\n"
                        continuation.resume(returning: ok)
                    }.resume()
                }
            }()
            results.append(.init(
                name: "proxy-ping",
                severity: pingResult ? .ok : .fail,
                detail: pingResult ? "200 ok" : "no response from \(pingURL.absoluteString)"
            ))
        }

        // Check 7: apiKeyHelper not set in ~/.claude/settings.json
        let claudeSettings = (NSString(string: "~/.claude/settings.json").expandingTildeInPath)
        if FileManager.default.fileExists(atPath: claudeSettings) {
            if let data = try? Data(contentsOf: URL(fileURLWithPath: claudeSettings)),
               let obj = try? JSONSerialization.jsonObject(with: data) as? [String: Any],
               obj["apiKeyHelper"] != nil
            {
                results.append(.init(name: "claude-apikeyhelper-absent", severity: .fail, detail: "apiKeyHelper set in \(claudeSettings) — incompatible with IRIS (Annex A.11)"))
            } else {
                results.append(.init(name: "claude-apikeyhelper-absent", severity: .ok, detail: "not set"))
            }
        } else {
            results.append(.init(name: "claude-apikeyhelper-absent", severity: .ok, detail: "no settings.json"))
        }

        emit(results)
        if results.contains(where: { $0.severity == .fail }) {
            throw ExitCode(IrisExitCode.logicError)
        }
    }

    private func emit(_ results: [CheckResult]) {
        if json {
            let encoder = JSONEncoder()
            encoder.outputFormatting = [.prettyPrinted, .sortedKeys]
            if let data = try? encoder.encode(results),
               let text = String(data: data, encoding: .utf8)
            {
                print(text)
            }
        } else {
            for r in results {
                print("[\(r.severity.rawValue)]  \(r.name)  — \(r.detail)")
            }
        }
    }
}
```

```swift
final class CLILogsFollowTests: XCTestCase {
    var harness: CLIDaemonHarness!

    override func setUpWithError() throws {

        try harness.start()
    }
```
