# Phase 5.1 — CLI Core Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Livrer le binaire `iris` câblé sur les RPCs daemon existants (`secret.*`, `daemon.*`, `ca.*`, `config.get`) avec UX cohérente (texte + `--json`), exit codes uniformes et invariant "aucune valeur secrète n'apparaît dans la sortie".

**Architecture:** Pattern racine `IrisCLI: AsyncParsableCommand` + sous-commandes dans `Sources/iris/Commands/`. Helpers réutilisables (`ConnectionOptions`, formatter, stdin reader) dans `Sources/iris/Support/` ; logique pure testable hostée dans `Sources/IrisKit/CLISupport/` pour pouvoir être testée hors processus iris (Swift Package Manager ne permet pas d'importer un executable target depuis un test target). Sous-commandes hors-scope Phase 5.1 (`rule.*`, `config reload`) exposées en stubs `EX_USAGE` 64.

**Tech Stack:** Swift 5.9+, `swift-argument-parser` 1.3+, `swift-nio` 2.65 (via `AdminClient`), `XCTest`. Cible macOS 13+. Strict concurrency.

**Source spec:** `docs/superpowers/specs/2026-05-25-phase-5-cli-design.md` §4 (pattern commun) + §5.1.

```swift
import Foundation

/// Locates executables produced by `swift build` so integration tests can
/// spawn them via `Process`. Resolves the directory that contains the
/// running `.xctest` bundle — SwiftPM places test bundles in the same
/// products directory as executables (`debug/` or `release/`).
enum ExecutableLocator {
    static func url(forProduct name: String) -> URL {
        for bundle in Bundle.allBundles where bundle.bundlePath.hasSuffix(".xctest") {
            return bundle.bundleURL
                .deletingLastPathComponent()
                .appendingPathComponent(name)
        }
        fatalError("XCTest bundle not found; cannot locate \(name)")
    }

    static var iris: URL { url(forProduct: "iris") }
    static var irisd: URL { url(forProduct: "irisd") }
}
```

```swift
import Foundation

/// Pure text rendering helpers shared between the `iris` CLI and any other
/// tool that needs the same column-aligned output. No I/O, no globals —
/// every function takes its inputs as values and returns a `String`.
public enum TextFormatter {
    /// Render a list of rows as a column-aligned table. Returns header-only
    /// when `rows` is empty. Columns are separated by two spaces and padded
    /// with trailing spaces so every line has identical width.
    public static func table(headers: [String], rows: [[String]]) -> String {
        let widths: [Int] = headers.indices.map { col in
            var width = headers[col].count
            for row in rows where col < row.count {
                width = max(width, row[col].count)
            }
            return width
        }
        let allRows = [headers] + rows
        let lines = allRows.map { row -> String in
            row.indices.map { col in
                let cell = col < row.count ? row[col] : ""
                let width = col < widths.count ? widths[col] : 0
                return cell.padding(toLength: width, withPad: " ", startingAt: 0)
            }
            .joined(separator: "  ")
            .trimmingCharacters(in: .init(charactersIn: " "))
        }
        return lines.joined(separator: "\n")
    }

    /// Single-line summary of a `DaemonStatus`.
    public static func status(_ status: DaemonStatus) -> String {
        let s = status.stats
        return "pid=\(status.pid) uptime=\(uptime(seconds: status.uptimeS)) "
            + "version=\(status.version) req=\(s.reqTotal) sub=\(s.subTotal) "
            + "exfil=\(s.exfilBlockedTotal) err=\(s.errorsTotal)"
    }

    /// Human-friendly uptime ("1d2h", "5m30s", "12s") — keeps the two
    /// most-significant units, drops trailing zero units, always non-empty.
    public static func uptime(seconds total: UInt64) -> String {
        let d = total / 86_400
        let h = (total % 86_400) / 3_600
        let m = (total % 3_600) / 60
        let s = total % 60
        var parts: [String] = []
        if d > 0 { parts.append("\(d)d") }
        if h > 0 { parts.append("\(h)h") }
        if d == 0 && m > 0 { parts.append("\(m)m") }
        if d == 0 && h == 0 && s > 0 { parts.append("\(s)s") }
        if parts.isEmpty { return "0s" }
        return parts.prefix(2).joined()
    }
}
```

```swift
import Foundation

public enum SecretReaderError: Error, Equatable, LocalizedError {
    case emptyValue

    public var errorDescription: String? {
        switch self {
        case .emptyValue:
            return "Secret value cannot be empty or whitespace-only"
        }
    }
}

/// Pure secret-value normalization. Strips at most one trailing `\n`
/// (LF) to make stdin pipes ergonomic without surprising callers that
/// pass binary data. Refuses whitespace-only inputs to catch obvious
/// mistakes (heredoc with empty line, accidental Enter at prompt).
public enum SecretReader {
    public static func read(from data: Data) throws -> Data {
        var trimmed = data
        if trimmed.last == 0x0A {
            trimmed = trimmed.dropLast()
        }
        let isWhitespaceOnly: Bool = {
            guard let text = String(data: trimmed, encoding: .utf8) else { return false }
            return text.allSatisfy(\.isWhitespace)
        }()
        if trimmed.isEmpty || isWhitespaceOnly {
            throw SecretReaderError.emptyValue
        }
        return trimmed
    }
}
```

```swift
import ArgumentParser
import Foundation
import IrisKit
import NIOPosix

/// `ExitCode` map shared by every `iris` subcommand. Mirrors design §4.2.
enum IrisExitCode {
    static let success: Int32 = 0
    static let logicError: Int32 = 1
    static let daemonUnreachable: Int32 = 2
    static let ioError: Int32 = 3
    static let usage: Int32 = 64  // EX_USAGE
}

/// Surfaces a daemon-unreachable failure with the canonical message before
/// re-throwing as `ExitCode` to short-circuit ArgumentParser.
struct DaemonUnreachable: Error, CustomStringConvertible {
    let socketPath: String
    var description: String {
        "irisd not running. Try: launchctl kickstart -k gui/$UID/io.iris.daemon\n"
            + "  socket: \(socketPath)"
    }
}

/// Option group shared by every subcommand that talks to the daemon.
struct ConnectionOptions: ParsableArguments {
    @Option(
        name: .customLong("socket-path"),
        help: "Path to the admin Unix socket (defaults to ~/Library/Application Support/iris/admin.sock)."
    )
    var socketPath: String?

    @Option(
        name: .customLong("config-path"),
        help: "Read socket path from the TOML config file at this path instead of the default."
    )
    var configPath: String?

    /// Resolves the effective socket path: `--socket-path` > `--config-path` >
    /// `Config.default.broker.adminSocket`.
    func resolvedSocketPath() throws -> String {
        if let explicit = socketPath, !explicit.isEmpty {
            return (explicit as NSString).expandingTildeInPath
        }
        if let cfgPath = configPath {
            let cfg = try ConfigLoader.loadTOML(atPath: cfgPath)
            return cfg.broker.expandedAdminSocket
        }
        return Config.default.broker.expandedAdminSocket
    }
}

/// Wraps the lifecycle of an `AdminClient`. Translates connection failures
/// into `DaemonUnreachable` so callers can return `ExitCode(2)`.
func withAdminClient<T: Sendable>(
    _ options: ConnectionOptions,
    body: (AdminClient) async throws -> T
) async throws -> T {
    let path = try options.resolvedSocketPath()
    let group = MultiThreadedEventLoopGroup(numberOfThreads: 1)
    let client = AdminClient(socketPath: path, group: group)
    do {
        let result = try await body(client)
        try await group.shutdownGracefully()
        return result
    } catch {
        try? await group.shutdownGracefully()
        if let adminErr = error as? AdminClientError, case .connectFailed = adminErr {
            throw DaemonUnreachable(socketPath: path)
        }
        throw error
    }
}
```

```swift
import Foundation
import IrisKit

enum Output {
    /// Print human-readable text, or JSON-encoded `value`, depending on
    /// `json`. `value` must be `Encodable` and is encoded with sorted keys +
    /// pretty-printing (cohérent avec wire format Phase 3 : snake_case via
    /// les `CodingKeys` explicites sur chaque type).
    static func print<E: Encodable>(
        humanText: @autoclosure () -> String,
        jsonValue: E,
        json: Bool
    ) throws {
        if json {
            let encoder = JSONEncoder()
            encoder.outputFormatting = [.prettyPrinted, .sortedKeys, .withoutEscapingSlashes]
            encoder.dateEncodingStrategy = .iso8601
            let data = try encoder.encode(jsonValue)
            if let text = String(data: data, encoding: .utf8) {
                Swift.print(text)
            }
        } else {
            Swift.print(humanText())
        }
    }

    /// Convenience for commands that don't produce a structured payload
    /// in `--json` mode (acks, no-ops). Emits `{"ok": true, "message": "..."}`
    /// when `json == true`.
    static func ack(message: String, json: Bool) throws {
        struct Ack: Encodable {
            let ok: Bool
            let message: String
        }
        try print(humanText: message, jsonValue: Ack(ok: true, message: message), json: json)
    }
}
```

        @Argument(help: "Secret name (used in placeholders as {{kc:NAME}}).")

        @Option(name: .customLong("value"), help: "Inline value (visible in shell history — prefer --value-from-stdin).")
        var inlineValue: String?

        @Flag(name: .customLong("json"), help: "Emit JSON output.")
        var json: Bool = false

        mutating func run() async throws {

                .split(separator: ",", omittingEmptySubsequences: true)

                value = try SecretReader.read(from: bytes)

                try await client.call(
                    .secretAdd,
                    params: SecretAddParams(name: name, allowedHosts: hosts, value: value),
                    returning: Secret.self

/// Reads a line from a TTY without echoing characters. Uses
/// `readpassphrase(3)` (Darwin libc) instead of `getpass(3)` because the
/// latter truncates silently at `PASS_MAX` (128 chars on macOS) — fatal
/// for long credentials (AWS session tokens, OAuth, long API keys).
/// Returns `nil` if no TTY is available or the read fails.
///
/// Note : `readpassphrase` n'est pas exposée par défaut via `import Darwin`.
/// L'engineer doit ajouter un module shim — créer
/// `Sources/CReadPassphrase/include/CReadPassphrase.h` :
/// ```c
/// #include <readpassphrase.h>
/// ```
/// puis un `module.modulemap` adjacent et déclarer un target system library
/// `CReadPassphrase` dans `Package.swift`, dépendance de `iris`. Importer
/// `CReadPassphrase` dans ce fichier.
private func readSecretFromTTY(prompt: String) -> Data? {
    var buf = [CChar](repeating: 0, count: 1024)
    guard readpassphrase(prompt, &buf, buf.count, RPP_REQUIRE_TTY) != nil else {
        return nil
    }
    return Data(String(cString: buf).utf8)
}

```swift
import Foundation
import XCTest

/// Spawns an ephemeral `irisd` instance with in-memory stores and provides
/// a helper to invoke the `iris` binary against it. Cleans up the temp
/// directory + admin socket on `tearDown`.
final class CLIDaemonHarness {
    let tmpDir: URL
    let configPath: String
    let adminSocket: String
    private(set) var process: Process?

    init() throws {
        let base = FileManager.default.temporaryDirectory
            .appendingPathComponent("iris-cli-tests-\(UUID().uuidString)")
        try FileManager.default.createDirectory(at: base, withIntermediateDirectories: true)
        self.tmpDir = base
        self.adminSocket = base.appendingPathComponent("admin.sock").path
        self.configPath = base.appendingPathComponent("iris.toml").path
        let toml = """
            [broker]
            listen = "127.0.0.1:0"
            admin_socket = "\(self.adminSocket)"

            [policy]
            on_exfil_attempt = "block_only"
            """
        try toml.write(toFile: self.configPath, atomically: true, encoding: .utf8)
    }

    func start() throws {
        let process = Process()
        process.executableURL = ExecutableLocator.irisd
        process.arguments = [
            "--foreground",
            "--config-path", configPath,
            "--in-memory-secrets",
            "--in-memory-ca",
        ]
        process.standardOutput = Pipe()
        process.standardError = Pipe()
        try process.run()
        self.process = process
        try waitForSocketReady(timeout: 5.0)
    }

    func stop() {
        process?.terminate()
        process?.waitUntilExit()
        process = nil
        try? FileManager.default.removeItem(at: tmpDir)
    }

    private func waitForSocketReady(timeout: TimeInterval) throws {
        let deadline = Date().addingTimeInterval(timeout)
        while Date() < deadline {
            if FileManager.default.fileExists(atPath: adminSocket) {
                Thread.sleep(forTimeInterval: 0.05)  // let bind permissions settle
                return
            }
            Thread.sleep(forTimeInterval: 0.05)
        }
        throw CLIDaemonHarnessError.timeout("admin socket not ready at \(adminSocket)")
    }

    /// Run `iris <args>` against the harness daemon. Returns (stdout, stderr, exitCode).
    @discardableResult
    func runIris(
        _ args: [String],
        stdin: Data? = nil,
        timeout: TimeInterval = 10.0
    ) throws -> (stdout: String, stderr: String, code: Int32) {
        let process = Process()
        process.executableURL = ExecutableLocator.iris
        process.arguments = ["--socket-path", adminSocket] + args
        let outPipe = Pipe()
        let errPipe = Pipe()
        process.standardOutput = outPipe
        process.standardError = errPipe
        let inPipe = Pipe()
        if let stdin = stdin {
            process.standardInput = inPipe
        }
        try process.run()
        if let stdin = stdin {
            inPipe.fileHandleForWriting.write(stdin)
            try inPipe.fileHandleForWriting.close()
        }
        process.waitUntilExit()
        let out = String(data: outPipe.fileHandleForReading.readDataToEndOfFile(), encoding: .utf8) ?? ""
        let err = String(data: errPipe.fileHandleForReading.readDataToEndOfFile(), encoding: .utf8) ?? ""
        return (out, err, process.terminationStatus)
    }
}

enum CLIDaemonHarnessError: Error {
    case timeout(String)
}
```

```swift
        XCTAssertNoLeak(rotate, secret: newValue)
        XCTAssertNoLeak(rotate, secret: secretValue)

        // rm with --yes
        let rm = try harness.runIris(["secret", "rm", "foo", "--yes"])
        XCTAssertEqual(rm.code, 0, "stderr=\(rm.stderr)")
        let listAfter = try harness.runIris(["secret", "list"])
        XCTAssertFalse(listAfter.stdout.contains("foo"))

    private func XCTAssertNoLeak(
        _ result: (stdout: String, stderr: String, code: Int32),
```
